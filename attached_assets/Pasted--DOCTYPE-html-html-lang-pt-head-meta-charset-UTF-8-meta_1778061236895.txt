// script.js
// Chave no localStorage
const LS_KEY = "citations";

// Utilidades de armazenamento
function loadAll() {
  const raw = localStorage.getItem(LS_KEY);
  try {
    const arr = raw ? JSON.parse(raw) : [];
    // Garantir array válido
    return Array.isArray(arr) ? arr : [];
  } catch {
    return [];
  }
}

function saveAll(arr) {
  localStorage.setItem(LS_KEY, JSON.stringify(arr));
}

function genId() {
  return `${Date.now()}-${Math.random().toString(36).slice(2, 8)}`;
}

// Validação e conflitos
function isValidAppointment({ client_name, service, date, time }) {
  return (
    client_name && client_name.trim() &&
    service && service.trim() &&
    date && date.trim() &&
    time && time.trim()
  );
}

function hasConflict(arr, { date, time }, excludeId = null) {
  return arr.some(item => item.date === date && item.time === time && item.id !== excludeId);
}

// Ordenação cronológica: por data, depois hora
function sortByChronology(arr) {
  return arr.slice().sort((a, b) => {
    const dA = a.date;
    const dB = b.date;
    if (dA < dB) return -1;
    if (dA > dB) return 1;
    if (a.time < b.time) return -1;
    if (a.time > b.time) return 1;
    return 0;
  });
}

// Highlight do dia atual
function isToday(dateStr) {
  const today = new Date();
  const d = new Date(dateStr + "T00:00:00");
  return (
    d.getFullYear() === today.getFullYear() &&
    d.getMonth() === today.getMonth() &&
    d.getDate() === today.getDate()
  );
}

// Helpers de DOM
function qs(sel) { return document.querySelector(sel); }
function qsa(sel) { return Array.from(document.querySelectorAll(sel)); }
function setMsg(text) {
  const el = qs("#msg");
  if (el) el.textContent = text || "";
}

// Página Agendar: criação/edição
function initAgendar() {
  const form = qs("#form-agendar");
  const btn = qs("#btn-confirmar");
  if (!form || !btn) return;

  // Se veio com ?id= para edição, carregar dados
  const params = new URLSearchParams(location.search);
  const editId = params.get("id");

  if (editId) {
    const arr = loadAll();
    const item = arr.find(x => x.id === editId);
    if (item) {
      qs("#nome").value = item.client_name;
      qs("#servico").value = item.service;
      qs("#data").value = item.date;
      qs("#hora").value = item.time;
      setMsg("Editando compromisso existente.");
    } else {
      setMsg("ID não encontrado. Criando novo compromisso.");
    }
  }

  btn.addEventListener("click", () => {
    const client_name = qs("#nome").value.trim();
    const service = qs("#servico").value.trim();
    const date = qs("#data").value;
    const time = qs("#hora").value;

    const payload = { client_name, service, date, time };

    if (!isValidAppointment(payload)) {
      setMsg("Preencha nome, serviço, data e hora.");
      return;
    }

    const arr = loadAll();

    // Evitar duplicados na mesma data/hora (exceto o próprio registro ao editar)
    if (hasConflict(arr, payload, editId || null)) {
      setMsg("Já existe um compromisso na mesma data e hora.");
      return;
    }

    if (editId) {
      // Atualizar
      const idx = arr.findIndex(x => x.id === editId);
      if (idx >= 0) {
        arr[idx] = {
          ...arr[idx],
          client_name,
          service,
          date,
          time,
          updatedAt: new Date().toISOString(),
        };
        saveAll(arr);
        setMsg("Compromisso atualizado com sucesso.");
      } else {
        setMsg("Não foi possível atualizar: ID não encontrado.");
        return;
      }
    } else {
      // Criar
      const newItem = {
        id: genId(),
        client_name,
        service,
        date,
        time,
        createdAt: new Date().toISOString(),
      };
      arr.push(newItem);
      saveAll(arr);
      setMsg("Compromisso criado com sucesso.");
    }

    // Redirecionar para a lista após 600ms
    setTimeout(() => { location.href = "comprimisso.html"; }, 600);
  });
}

// Página Lista (comprimisso.html): render, filtros, ações
function initLista() {
  const tbody = qs("#lista-compromissos");
  if (!tbody) return;

  const filtroData = qs("#filtro-data");
  const filtroServico = qs("#filtro-servico");
  const btnLimpar = qs("#limpar-filtros");

  function render() {
    const arr = sortByChronology(loadAll());
    const fd = filtroData?.value || "";
    const fs = filtroServico?.value || "";

    const filtrados = arr.filter(item => {
      const okData = fd ? item.date === fd : true;
      const okServ = fs ? item.service === fs : true;
      return okData && okServ;
    });

    tbody.innerHTML = "";

    if (filtrados.length === 0) {
      tbody.innerHTML = `<tr><td colspan="5">Nenhum compromisso encontrado.</td></tr>`;
      return;
    }

    filtrados.forEach(item => {
      const tr = document.createElement("tr");

      // Destaque para compromissos do dia
      if (isToday(item.date)) {
        tr.style.backgroundColor = "#ffe8a1"; // leve destaque
      }

      tr.innerHTML = `
        <td>${item.client_name}</td>
        <td>${item.service}</td>
        <td>${item.date}</td>
        <td>${item.time}</td>
        <td>
          <button class="editar" data-id="${item.id}">Editar</button>
          <button class="cancelar" data-id="${item.id}">Cancelar</button>
        </td>
      `;
      tbody.appendChild(tr);
    });

    // Bind de ações
    qsa("button.editar").forEach(btn => {
      btn.onclick = () => {
        const id = btn.getAttribute("data-id");
        location.href = `agendar.html?id=${encodeURIComponent(id)}`;
      };
    });

    qsa("button.cancelar").forEach(btn => {
      btn.onclick = () => {
        const id = btn.getAttribute("data-id");
        const arr = loadAll();
        const novo = arr.filter(x => x.id !== id);
        saveAll(novo);
        setMsg("Compromisso cancelado.");
        render();
      };
    });
  }

  // Filtros
  if (filtroData) filtroData.addEventListener("change", render);
  if (filtroServico) filtroServico.addEventListener("change", render);
  if (btnLimpar) btnLimpar.addEventListener("click", () => {
    if (filtroData) filtroData.value = "";
    if (filtroServico) filtroServico.value = "";
    render();
  });

  render();
}

// Inicialização por página
document.addEventListener("DOMContentLoaded", () => {
  initAgendar();
  initLista();
});
